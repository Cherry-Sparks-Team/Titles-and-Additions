IlkhanMaleChange = {
	type = character

	text = {
		trigger = {
			OR = {
				any_held_title = {
					count >= 2
					tier = tier_empire
				}
				any_held_title = {
					count >= 3
					tier = tier_kingdom
				}
				any_vassal = {
					highest_held_title_tier >= tier_kingdom
					count >= 3
				}
				any_spouse = {
					OR = {
						any_held_title = {
							count >= 2
							tier = tier_empire
						}
						any_held_title = {
							count >= 3
							tier = tier_kingdom
						}
						any_vassal = {
							highest_held_title_tier >= tier_kingdom
							count >= 3
						}
					}
				}
			}
		}
		localization_key = ilkhan
	}
	text = {
		fallback = yes
		localization_key = emperor_feudal_male_mongolic_default
	}
}
