# All on-actions in this file provide scope:transfer_type unless otherwise noted
# scope:transfer_type has the following permutations:
# flag:conquest
# flag:conquest_holy_war
# flag:conquest_claim
# flag:conquest_populist
# flag:inheritance
# flag:abdication
# flag:destroyed
# flag:created

# flag:granted

# flag:faction_demand
# flag:swear_fealty

# A title is destroyed
# root = the holder before destruction
# scope:landed_title = the title that is being destroyed
# Does *not* provide scope:transfer_type
on_action_ATM_title_destroyed_e_persia = {
	trigger = {
		scope:landed_title = {
			this = title:e_persia
		}
	}
	effect = {
		title:e_persia = {
			reset_title_name = yes
		}
	}
}

# A title is lost by a character
# root = the old holder
# scope:title = the title that changes hands
# scope:new_holder = new holder.
on_action_ATM_title_lost_caliphate_imamate_e_persia = {
	trigger = {
		has_title = title:e_persia
	}
	on_actions = {
		on_action_ATM_title_change_e_persia
	}
}

# A title is transferred to a new character
# root = the new holder
# scope:title = the title that changes hands
# scope:previous_holder = previous holder. Might be dead
on_action_ATM_title_gain_e_persia = {
	trigger = {
		scope:title = {
			this = title:e_persia
		}
	}
	on_actions = {
		on_action_ATM_title_change_e_persia
	}
}
